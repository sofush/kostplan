@startuml
scale 1920*1080
entity User {
	{field} *username : VARCHAR(20)
	password : TEXT
	email_address : TEXT
	phone_number : TEXT
	activity_level : INTEGER
	weight_goal : INTEGER
	role : INTEGER
	name : TEXT
	male : BOOLEAN
	weight : INTEGER
	dob : DATE
	height : INTEGER
}
entity Day {
	*date : DATE
	*{field} username : VARCHAR(20)
	breakfast : INTEGER <<FK>>
	lunch : INTEGER <<FK>>
	dinner : INTEGER <<FK>>
}
entity Recipe {
	*id : INTEGER
	week : INTEGER
	title : TEXT
	description : TEXT
	instructions : TEXT
	time : TEXT
}
entity RecipeImage {
	*recipe : INTEGER <<FK>>
	bytes : LONGBLOB
}
entity Ingredient {
	*id : INTEGER
	recipe : INTEGER <<FK>>
	name : TEXT
	quantity : DOUBLE
	unit : INTEGER <<FK>>
	calories : DOUBLE
}
entity Unit {
	*id : INTEGER
	{field} unit : VARCHAR(100)
}
User ||.r.o{ Day
Day ||..o{ Recipe
Recipe ||.r.|| RecipeImage
Recipe ||..|{ Ingredient
Ingredient ||.l.|| Unit
@enduml
